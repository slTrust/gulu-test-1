#### 高级一点的测试用例 button.test.js

```
const expect = chai.expect;
import Vue from 'vue'
import Button from '../src/button'

Vue.config.productionTip = false
Vue.config.devtools = false

describe('Button', () => {
    it('存在.', () => {
        expect(Button).to.be.ok
    })
    it('可以设置icon.', () => {
        const Constructor = Vue.extend(Button)
        const vm = new Constructor({
            propsData: {
                icon: 'settings'
            }
        }).$mount()
        const useElement = vm.$el.querySelector('use')
        expect(useElement.getAttribute('xlink:href')).to.equal('#i-settings')
        vm.$destroy()
    })
    it('可以设置loading.', () => {
        const Constructor = Vue.extend(Button)
        const vm = new Constructor({
            propsData: {
                icon: 'settings',
                loading: true
            }
        }).$mount()
        const useElements = vm.$el.querySelectorAll('use')
        expect(useElements.length).to.equal(1)
        expect(useElements[0].getAttribute('xlink:href')).to.equal('#i-loading')
        vm.$destroy()
    })
    it('icon 默认的 order 是 1', () => {
        const div = document.createElement('div')
        document.body.appendChild(div)
        const Constructor = Vue.extend(Button)
        const vm = new Constructor({
            propsData: {
                icon: 'settings',
            }
        }).$mount(div)
        const icon = vm.$el.querySelector('svg')
        expect(getComputedStyle(icon).order).to.eq('1')
        vm.$el.remove()
        vm.$destroy()
    })
    it('设置 iconPosition 可以改变 order', () => {
        const div = document.createElement('div')
        document.body.appendChild(div)
        const Constructor = Vue.extend(Button)
        const vm = new Constructor({
            propsData: {
                icon: 'settings',
                iconPosition: 'right'
            }
        }).$mount(div)
        const icon = vm.$el.querySelector('svg')
        expect(getComputedStyle(icon).order).to.eq('2')
        vm.$el.remove()
        vm.$destroy()
    })
    it('点击 button 触发 click 事件', () => {
        const Constructor = Vue.extend(Button)
        const vm = new Constructor({
            propsData: {
                icon: 'settings',
            }
        }).$mount()

        const callback = sinon.fake();
        vm.$on('click', callback)
        vm.$el.click()
        expect(callback).to.have.been.called

    })
})
```

### 如何知道怎么写测试用例呢？

```
// 期待 xxx = yyy
expect(xxx).to.equal(yyy)
// 简写
expect(xxx).to.eq(yyy)

// 期待 arr 是一个数组
expect(arr instanceof Array).to.equal(true)
```

> 具体语法 chaijs官网

[https://www.chaijs.com/](https://www.chaijs.com/)


- 选中expect 点击 Visit Expect Guide 专家指南

```
// 期待 xxx = yyy
expect(xxx).to.equal(yyy)
expect(xxx).to.be.equal(yyy)

// be是虚词 可以省略的


//not
// 期待 xxx 不存在
expect(xxx).to.be.not.ok 


// deep 深相等

expect([1,2]).to.eq([1,2]) // 失败 比较的是地址
expect([1,2]).to.deep.eq([1,2]) //成功 

// own 是否有一个属性
expect({a:1}).to.have.own.property('a')
// 看一个对象是自己的属性 还是继承的属性
expect({a:1}).to.have.own.property('a').but.not.own.property('b')



// NaN
expect(NaN).to.equal(NaN)  // 失败 NaN 是不等于NaN的

expect(NaN).to.be.NaN // 成功
```
